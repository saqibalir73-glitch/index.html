// download.html ke script mein ye wala connection check lagayein async function loadData() { const params = new URLSearchParams(window.location.search); const mId = params.get('id'); try { const res = await fetch('news.json'); const data = await res.json(); // ID ko match karne ka naya aur pakka tarikha const movie = data.news_items.find(m => String(m.id) === String(mId)); if(movie) { document.getElementById('title').innerText = movie.title; document.getElementById('poster').src = movie.img; } else { document.getElementById('title').innerText = "Movie Data Syncing... Please Wait"; } } catch(e) { console.log("Json Error"); } }
